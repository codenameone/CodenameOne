package com.codename1.impl.android;

//import com.codename1.tests.webpush.WebPushTestStub;

/**
 * This class will be modified by the build server to return the correct Application stub class
 * for the built application.  This is used to try to minimize the amount of code that needs to be
 * generated by the server.  Classes in the android implementation can use the StubUtil static methods
 * instead of needing to have code injected by the build server.  
 * 
 * The build server will replace all instances of "" (without quotes) with "" (without quotes).
 * And  "com.codenameone.examples.hellocodenameone.HelloCodenameOneStub" (without quotes) with the fully qualified class name of the app stub.
 * @author shannah
 */
public class StubUtil {

    static boolean appIsRunning() {
        if (true) {
            return com.codenameone.examples.hellocodenameone.HelloCodenameOneStub.isRunning();
        }
        if (true) {
            //return WebPushTestStub.isRunning();
        }
        return false;
    }


    static Class getAppStubClass() {
        if (true) {
            return com.codenameone.examples.hellocodenameone.HelloCodenameOneStub.class;
        }
        if (true) {
            //return WebPushTestStub.class;
        }
        return null;

    }

    static Object getMain() {
        if (true) {
            return com.codenameone.examples.hellocodenameone.HelloCodenameOneStub.getAppInstance();
        }
        if (true) {
            //return WebPushTestStub.getAppInstance();
        }

        return null;
    }
}
