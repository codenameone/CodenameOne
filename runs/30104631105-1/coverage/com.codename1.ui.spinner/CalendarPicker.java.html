package com.codename1.ui.spinner;

import com.codename1.ui.Container;
import com.codename1.ui.layouts.BorderLayout;

import java.util.Date;

/// A Calendar Picker widget.  Used by the picker.
///
/// @author Steve Hannah
class CalendarPicker extends Container implements InternalPickerWidget {
    private com.codename1.ui.Calendar calendar;

    public CalendarPicker() {
        init();
    }

    private void init() {
        setLayout(new BorderLayout());
        calendar = new com.codename1.ui.Calendar();
        add(BorderLayout.CENTER, calendar);

    }

    @Override
    public Object getValue() {
        return calendar.getDate();
    }

    @Override
    public void setValue(Object value) {
        if (value == null) {
            // A null value (e.g. Picker.setDate(null) while the popup is on
            // screen) has no meaningful representation; leave the current
            // selection untouched instead of NPE'ing. Mirrors TimeSpinner3D's
            // null tolerance. #5069
            return;
        }
        Date dt = (Date) value;
        calendar.setDate(dt);

    }

}
